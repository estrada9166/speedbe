'use strict'

const jwt = require('jsonwebtoken');

const newToken = (req, res) => {
  const data = req.body;
  const jwtSecret = process.env.JWTKEY;
  const token = jwt.sign({ data }, jwtSecret, { expiresIn: '' + 1000 * 60 * 60 * 24 * 365 });
  res.json({
    success: true,
    type: 'Bearer',
    token
  })
}

module.exports = { newToken }
